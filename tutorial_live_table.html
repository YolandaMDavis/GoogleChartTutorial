<script lang="javascript"> 
var tableChart; 
var gaugeChart;
var refreshInterval = 10000;
var counter = 0;
var tweetNumber = 100;

 function drawCharts(dataTable,gaugeTable){   
    tableChart.clearChart(); 	
	tableChart.draw(dataTable, {allowHtml: true, page: 'enable', pageSize: 10});
	gaugeChart.draw(gaugeTable,  {width: 400, height: 120, redFrom: 90, redTo: 100,yellowFrom:75, yellowTo: 90,minorTicks: 5, animation:{duration: 400,easing: 'out'}});	
 }
 
function populateChart(data){	
 
	var dataTable = new google.visualization.DataTable();
	dataTable.addColumn('string', 'Twitter User');
	dataTable.addColumn('string', 'Tweet Text');
	dataTable.addColumn('datetime', 'Tweet Date');	
	
	$.each(data.results,function(i,value){
	  var link = 'https://twitter.com/#!/'+value.from_user+'/status/'+value.id_str;
	  var htmlText = '<a href="'+link+'" target="_blank">'+value.text+'</a>';	  
      dataTable.addRow([value.from_user,htmlText,new Date(value.created_at)]);	  
	});	

	var gaugeTable = google.visualization.arrayToDataTable([['Label', 'Value'],['Tweets', dataTable.getNumberOfRows()]]);		
	drawCharts(dataTable,gaugeTable);
}

function getJsonData(){
	var bcOpenDataUrl = 'http://search.twitter.com/search.json?q=BDPAConference&rpp='+tweetNumber+'&callback=populateChart';
	var script = '<script type="text/javascript" src="'+bcOpenDataUrl+'"/>';	
	$('head').append(script);                   
}

 function countDown(){	
	if(counter == 0){
		counter = refreshInterval/1000;
		getJsonData();
	}else{
		counter--;
	} 
	$("#secondSpan").html(counter);
 }
 
 google.load('visualization', '1.0', {'packages':['table','gauge']});

 $(function() { 		
	tableChart = new google.visualization.Table(document.getElementById('tableChartDisplay')); 	
	gaugeChart = new google.visualization.Gauge(document.getElementById('gaugeChartDisplay')); 	
	setInterval("countDown()",1000);
	$('#tweetNumber').html(tweetNumber);
 });	
 </script>
