Bar Chart

Importing Google API: 


	<!--In HTML Header -->
	<script type="text/javascript" src="http://www.google.com/jsapi"></script>

Chart Code: 


	//In Javascript

	google.load('visualization', '1.0', {'packages':['corechart']});
	 
	function drawChart(){
		var data = getDataTable();
		var options = {title: 'Yolanda\'s Favorite Cities', height: '300'};
		var chart = new google.visualization.BarChart(document.getElementById('chartDisplay'));
		chart.draw(data, options);
	}
