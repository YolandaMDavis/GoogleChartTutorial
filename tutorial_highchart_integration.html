//creating a namespace
var ymd = {};

ymd.HighChartLine = function(container) {
  this.containerElement = container;
}
    
ymd.HighChartLine.prototype.draw = function(data, options) {

   var chart = new Highcharts.Chart({

        chart: {
            renderTo: this.containerElement.id,
			type: 'column'
        },
		title: {
			text: options.title			
		},
    
        xAxis: {
            categories: convertDataToArray(data,0),           
        },
    
        series: [{
            data:convertDataToArray(data,1)
        }]

    }); 
    
};

function convertDataToArray(data, colIndex){

    var seriesArray = new Array();
    for (i = 0; i < data.getNumberOfRows(); i++){			  	     	
		  	seriesArray.push(data.getValue(i,colIndex));                		 
    }                       
    return seriesArray;
                           
}

function getDataTable(){

      var data = new google.visualization.DataTable();
      data.addColumn('string', 'Favorite Cities');
      data.addColumn('number', 'Rating');      
      data.addRows([
        ['Baltimore', 10],
        ['Atlanta', 9],
        ['Washington, D.C.', 8], 
        ['New York', 7],
        ['Dallas', 8]
      ]);                        
      return data;
}

    
function drawChart() {    

    var chartContainer = document.getElementById('highChartDisplay');
	try{
	
		var data = getDataTable();	
		data.sort(1);
		var view = new google.visualization.DataView(data)
		view.hideRows(0,1);
		var chart = new ymd.HighChartLine(chartContainer);
		chart.draw(view, {title:'HighCharts Integration Chart'});
		
	}
	catch(err){
		google.visualization.errors.addError(chartContainer,"An error occured while rendering the chart", err.message, {'showInTooltip': false});
	}
}

google.load("visualization", "1");
google.setOnLoadCallback(drawChart);
