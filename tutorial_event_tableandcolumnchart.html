google.load('visualization', '1.0', {'packages':['corechart','table']});
 
function drawCharts(){
	var data = getDataTable();
	var options = {title: 'Yolanda\'s Favorite Cities', height: '300'};
	
	var barChart = new google.visualization.BarChart(document.getElementById('barChartDisplay'));
	var table = new google.visualization.Table(document.getElementById('tableChartDisplay'));
	
	table.draw(data, null);
	barChart.draw(data, options);
	
	google.visualization.events.addListener(table, 'select',function(e) {
		 barChart.setSelection(table.getSelection());		 
	});
		
	google.visualization.events.addListener(barChart, 'select',function(e) {
		var selection = barChart.getSelection();
		var dataValue = data.getValue(selection[0].row, selection[0].column);
		var cityValue = data.getValue(selection[0].row, selection[0].column - 1);
		$("#chartMessage").html("The city of " + cityValue + " received a rating of " + dataValue + " out of 10");
		$("#chartMessage").dialog("open");		
	});
		
}

$(function() {

	$("#chartMessage" ).dialog({
			modal: true,
			autoOpen: false,
			buttons: {
				Ok: function() {
					$( this ).dialog( "close" );
				}
			}
		});
	
	drawCharts();
});
